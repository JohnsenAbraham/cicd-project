const express = require('express');
const app = express();

app.use(express.json());

// Root route
app.get('/', (req, res) => {
    res.status(200).json({ message: "CI/CD Pipeline Working" });
});

// Health check
app.get('/health', (req, res) => {
    res.status(200).json({ status: "OK" });
});

module.exports = app;

Infrastructure
⚡

GitHub Actions

Triggered on every push to main. Four sequential jobs: build, security, quality, and deploy — each gating the next.

🐳

Docker

Minimal node:18-alpine image. Production-only dependencies, auto-pushed to DockerHub on successful pipeline completion.

☁️

Netlify Serverless

Express app wrapped in serverless-http and deployed as a Netlify Function — zero cold-start overhead.

🧪

Jest + Supertest

Integration tests against the live Express router — root, health, and 404 coverage — with automatic coverage reports uploaded to SonarCloud.

🔍

SonarCloud

Continuous code quality and security analysis. Quality gate must pass before Docker build runs — bad code never ships.

🔐

Security Gate

npm audit --audit-level=high blocks on critical CVEs. Secrets managed via GitHub encrypted secrets, never hardcoded.

FROM node:18-alpine

WORKDIR /app

COPY package*.json ./
RUN npm ci --only=production   # lean image, no devDeps

COPY . .

EXPOSE 3000
CMD ["node", "src/server.js"]
